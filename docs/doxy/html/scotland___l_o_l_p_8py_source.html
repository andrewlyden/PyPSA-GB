199def probability_function(capacity, breakdwon_rate):
200 pdf = dict()
201 pdf[0] = 1
202
203 for i in range(capacity.shape[0]):
204 prob_down = dict()
205 prob_up = dict()
206
207 for key,value in pdf.items():
208 avail_value = np.float64(key)+capacity[i]
209 try:
210 prob_up[avail_value] += value * (1-breakdwon_rate[i])
211 except:
212 prob_up[avail_value] = value * (1-breakdwon_rate[i])
213 try:
214 prob_down[key] += value * breakdwon_rate[i]
215 except:
216 prob_down[key] = value * breakdwon_rate[i]
217 pdf = dict_add(prob_up,prob_down)
218 pdf = dict(sorted(pdf.items(),key=lambda d:d[0]))
219 xx = sorted(pdf)
220 yy = np.zeros(len(xx))
221 cdf = np.zeros(len(xx))
222 cdf_i = 0
223 for i in range(len(pdf)):
224 key = xx[i]
225 yy[i] = pdf[key]
226 cdf_i += yy[i]
227 cdf[i] = cdf_i
228 return xx, yy, pdf, cdf
