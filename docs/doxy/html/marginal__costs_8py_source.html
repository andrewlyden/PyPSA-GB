174 result.rename(
175 columns={'EU ETS (Euros/tonne)': 'EU ETS (Pounds/tonne)'},
176 inplace=True)

178 # now to calculate the carbon tax and total prices
179 # using conversion factors from 2021 DUKES data,
180 # this should probably change from year to year
181
182 # these are from
183 # https://www.parliament.uk/globalassets/documents/post/postpn_383-carbon-footprint-electricity-generation.pdf
184 coal_emission_factor = 846
185 # really should be different between OCGT and CCGT
186 gas_emission_factor = 488
187 # need better reference for oil
188 # https://www.jcm.go.jp/cl-jp/methodologies/68/attached_document2
189 oil_emission_factor = 533
190
191 result.loc[:, 'Gas carbon tax (p/kWh)'] = (
192 result['Carbon support price (Pounds/tonne)'] +
193 result['EU ETS (Pounds/tonne)']) * gas_emission_factor / 10000
194 result.loc[:, 'Coal carbon tax (p/kWh)'] = (
195 result['Carbon support price (Pounds/tonne)'] +
196 result['EU ETS (Pounds/tonne)']) * coal_emission_factor / 10000
197 result.loc[:, 'Oil carbon tax (p/kWh)'] = (
198 result['Carbon support price (Pounds/tonne)'] +
199 result['EU ETS (Pounds/tonne)']) * oil_emission_factor / 10000
200
201 # converting to £/MWh
202 result.loc[:, 'Gas'] = (
203 result['Gas carbon tax (p/kWh)'] +
204 result['Gas (p/kWh)']) * 10
205 result.loc[:, 'Coal'] = (
206 result['Coal carbon tax (p/kWh)'] +
207 result['Coal (p/kWh)']) * 10

248 future_result['High case']) * oil_emission_factor / 10000

250 # converting to £/MWh
251 # gas is in p/therm, divide by 29.3 * 0.5 = 14.65, assume elec efficiency 50%
252 # this gets us to p/kWh, then * 10 to get £/MWh
253 future_result.loc[:, 'Gas'] = (
