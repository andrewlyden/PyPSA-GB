64def offshore_wind_time_series(df, year, date_from, date_to):
65 """offshore wind time series download from renewables ninja
66
67 writes the timeseries to pickle file in relevant folder
68 note that only single year can be accessed
69
70 Parameters
71 ----------
72 df : dataframe
73 REPD data, should use date corrected version
74 year: int/str
75 Year of simulation
76 date_from: str
77 time series date from
78 date_to: str
79 time series date to
80
81 Returns
82 -------
83 """
84
85 # aim is to extract power series for
86 # offshore wind farms from renewables.ninja
87 df_offshore_wind = df.loc[df['Technology Type'] == 'Wind Offshore']
88 df_offshore_wind = df_offshore_wind.reset_index()
89
90 # check that data is available
91 if df_offshore_wind['No. of Turbines'].isnull().values.any() == True:
92 raise Exception("Number of turbines values must be provided, please fill in missing data in csv file")
93 if df_offshore_wind['Turbine Capacity (MW)'].isnull().values.any() == True:
94 raise Exception("Number of turbines values must be provided, please fill in missing data in csv file")
95
96 output_series = []
97 output_series_normalised = []
98 output_series_metadata = {}
99
100 length = df_offshore_wind.index
101 # length = range(3)
102 print(length, ' number of requests.')
103 for i in range(len(length)):
104
105 print(i)
106 print(df_offshore_wind['Site Name'][i])
107
108 # multiplied by 1000 to convert to kW
109 capacity = df_offshore_wind['Turbine Capacity (MW)'][i] * 1000
110
111 # check if height given
112 if df_offshore_wind['Height of Turbines (m)'][i] == 'NaN':
113 height = df_offshore_wind['Height of Turbines (m)'][i]
114 else:
115 # need to estimate height from turbine capacity using regression
116 height = wind_hub_height_to_capacity_regression(
117 capacity)
118
119 # height is limited to 150m
120 height = min(height, 150)
121
122 # need to figure out which turbine to use for power curve
123 # vestas has a wide range, let just use their ones
124 turbine = turbine_from_capacity(capacity)
125
126 # location parameters
127 lat = df_offshore_wind['lat'][i]
128 lon = df_offshore_wind['lon'][i]
129
130 print(lat, lon, date_from, date_to, capacity, height, turbine)
131
132 # request data from renewables ninja
133 wind = renewables_ninja_API.request_wind(
134 token, lat, lon, date_from, date_to,
135 capacity, height, turbine)
136
137 wind['data'] = wind['data'].rename(
138 columns={'electricity': df_offshore_wind['Site Name'][i]})
139 # multiply by number of wind turbines
140 wind['data'] = wind['data'] * df_offshore_wind['No. of Turbines'][i]
141 wind['normalised'] = wind['data'] / (capacity * df_offshore_wind['No. of Turbines'][i])
142 output_series.append(wind['data'])
143 output_series_normalised.append(wind['normalised'])
144 output_series_metadata[df_offshore_wind['Site Name'][i]] = wind['metadata']
145
146 # gathers all the time series into one dataframe
147 df_out = pd.concat(output_series, axis=1)
148 df_out_norm = pd.concat(output_series_normalised, axis=1)
149
150 # want to save these into pickle files
151 df_out.to_pickle("data/renewables/" + str(year) + "/offshore_time_series.pkl")
152 df_out_norm.to_pickle("data/renewables/" + str(year) + "/offshore_time_series_norm.pkl")
153 with open('data/renewables/' + str(year) + '/offshore_metadata.pkl', 'wb') as handle:
154 pickle.dump(
155 output_series_metadata, handle, protocol=pickle.HIGHEST_PROTOCOL)

331def wind_hub_height_to_capacity_regression(capacity):
332 """guesses hub height from capacity
333
334 Parameters
335 ----------
336 capacity : int/float
337 capacity of wind turbine
338
339 Returns
340 -------
341 hub_height : int
342 predicted hub height of wind turbine
343 """
344 file = 'data/renewables/wind_regression.csv'
345 df = pd.read_csv(file)
346 # values converts it into a numpy array
347 X = df['Rated Capacity'].values.reshape(-1, 1)
348 # -1 means that calculate the dimension of rows, but have 1 column
349 Y = df['Hub Height'].values.reshape(-1, 1)
350 linear_regressor = LinearRegression() # create object for the class
351 linear_regressor.fit(X, Y) # perform linear regression
352
353 # get value and reshape into single value and convert to int
354 return int(linear_regressor.predict([[capacity]])[:, 0][0])

357def turbine_from_capacity(capacity):
358 """guesses turbine type from capacity using closest capacity in list
359
360 Parameters
361 ----------
362 capacity : int/float
363 capacity of wind turbine
364
365 Returns
366 -------
367 turbine : str
368 predicted turbine type
369 """
370 turbine_data = {225: 'Vestas V29 225',
371 500: 'Vestas V39 500',
372 600: 'Vestas V44 600',
373 660: 'Vestas V47 660',
374 850: 'Vestas V52 850',
375 1000: 'NEG Micon NM60 1000',
376 1650: 'Vestas V66 1650',
377 1750: 'Vestas V66 1750',
378 1800: 'Vestas V80 1800',
379 2000: 'Vestas V90 2000',
380 3000: 'Vestas V90 3000',
381 3300: 'Vestas V112 3300',
382 # 3400: 'REpower 3 4M',
383 5000: 'REpower 5M',
384 6000: 'REpower 6M'}
385
386 turbine = turbine_data.get(capacity) or turbine_data[
387 min(turbine_data.keys(),
388 key=lambda key: abs(key - capacity))]
389 # turbine = turbine.replace(".", " ")
390 return turbine
