46 """reads the future demand profile data from Staffel et al - https://doi.org/10.1016/j.energy.2015.06.082

57 # this stuff can be used to get data from national grid csv files
58 # file1 = 'data/demand/DemandData_2005-2010.csv'
59 # file2 = 'data/demand/DemandData_2011-2016.csv'

72 # using espeni data set
73 file = '../data/demand/egy_7649_mmc1.xlsx'
74 df = pd.read_excel(file, sheet_name=None)
75 df_eload = df['ELOAD'].drop([0, 1, 2, 3, 4, 5])[['eLOAD Model (2050).2']].reset_index(drop=True) * 1000
76 df_eload.rename(columns={'eLOAD Model (2050).2': 'eLOAD'}, inplace=True)

80 # resample to half hour frequency
81 df_eload['eLOAD'] = pd.to_numeric(df_eload['eLOAD'])
82 df_eload = df_eload.resample('0.5h').interpolate('polynomial', order=2)
83
84 # add end value
85 df_new_eload = pd.DataFrame(
86 data=df_eload.tail(1).values,
87 columns=df_eload.columns,
88 index = pd.date_range(start='2050-12-31 23:30:00', end='2050-12-31 23:30:00', freq='0.5h'))
89 # add to existing dataframe
90 df_eload = pd.concat([df_eload, df_new_eload], sort=False)
91
92 return df_eload

337def read_regional_breakdown_load(scenario, year, networkmodel):
338
339 if scenario == 'Leading the Way':
340 df_regional = pd.read_excel('../data/FES2022/FES22_regional_peak_load_leading_the_way.xlsx', sheet_name=str(year), header=5)
341 elif scenario == 'Consumer Transformation':
342 df_regional = pd.read_excel('../data/FES2022/FES22_regional_peak_load_consumer_transformation.xlsx', sheet_name=str(year), header=5)
343 elif scenario == 'System Transformation':
344 df_regional = pd.read_excel('../data/FES2022/FES22_regional_peak_load_system_transformation.xlsx', sheet_name=str(year), header=5)
345 elif scenario == 'Falling Short':
346 df_regional = pd.read_excel('../data/FES2022/FES22_regional_peak_load_falling_short.xlsx', sheet_name=str(year), header=5)
347 # remove first row
348 df_regional = df_regional.iloc[1: , :].set_index('Name')
349 # delete all duplicates in index (removes electrolysis stuff which is zero for winter peak anyway)
350 df_regional = df_regional[~df_regional.index.duplicated(keep=False)]
351 # print(df_regional['P(Gross'])
352 df_regional = df_regional[['P(Gross)']]
353 df_regional = df_regional.loc[~(df_regional==0).all(axis=1)]
354
355 df_gsp_data = pd.read_csv('../data/FES2022/GSP_data.csv', encoding='cp1252', index_col=3)
356 df_gsp_data = df_gsp_data[['Latitude', 'Longitude']]
357 df_gsp_data.rename(columns={'Latitude': 'y', 'Longitude': 'x'}, inplace=True)
358
359 if networkmodel == 'Reduced':
360 from distance_calculator import map_to_bus as map_to
361 elif networkmodel == 'Zonal':
362 from allocate_to_zone import map_to_zone as map_to
363
364 df_gsp_data['Bus'] = map_to(df_gsp_data)
365 # now
366 GSP_to_bus = []
367 for GSP in df_regional.index:
368 GSP_to_bus.append(df_gsp_data['Bus'][GSP])
369
370 df_regional['Bus'] = GSP_to_bus
371 # list of buses
372 peak_bus = {}
373 for bus in df_regional['Bus'].unique():
374 peak_bus[bus] = df_regional.loc[df_regional['Bus'] == bus]['P(Gross)'].sum()
375
376 return peak_bus
