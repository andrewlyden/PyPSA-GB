33class copy_file:
34 def __init__(self, dir1, dir2):
35 dlist = os.listdir(dir1)
36 if not os.path.exists(dir2):
37 os.mkdir(dir2)
38 for f in dlist:
39 file1 = os.path.join(dir1, f)
40 file2 = os.path.join(dir2, f)
41 if os.path.isfile(file1):
42 shutil.copyfile(file1, file2)
43 if os.path.isdir(file1):
44 self.__init__(file1, file2)
